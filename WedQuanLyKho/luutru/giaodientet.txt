function postIndex($login,$value="")
{
    if(!isset($_POST[$login]))
        return $value;
    return trim($_POST[$login]);
}

    $stmt = $conn->prepare("SELECT * FROM taikhoan WHERE TenDangNhap = ? AND MatKhau = ?");

    $row = $stmt->fetch(PDO::FETCH_ASSOC);

    if ($row) 
    {
        // Lưu session
         $_SESSION['MaTK'] = $row['MaTK'];   
        $_SESSION["username"] = $row['TenDangNhap'];
        $_SESSION["MaNV"] = $row['MaNV']; // Lưu MaNV nếu cần

        exit();
